#
# Cách dùng:
#   giật tít            Viết thẳng từ, tin tức chứa "giật tít" sẽ bị lọc
#   /cờ bạc|cá cược/    Dùng /.../ để khớp nhiều từ cùng lúc (phân cách bằng |)

# ───────────────────────────────────────────────────────────────
#                        Khu Vực Định Nghĩa Nhóm Từ
# ───────────────────────────────────────────────────────────────
# Viết các từ khóa bạn muốn theo dõi vào đây.
# Mỗi nhóm từ cách nhau bằng một dòng trống, các từ khóa trong cùng nhóm có quan hệ "HOẶC".
#
# ┌─────────────────────────────────────────────────────────────┐
# │                  Tổng Quan Cú Pháp (Tham Khảo Nhanh)        │
# └─────────────────────────────────────────────────────────────┘
#
# Cú pháp từ khóa:
#   từ khóa           Từ khóa thường, khớp nếu tiêu đề chứa từ này
#   /regex/           Khớp bằng biểu thức chính quy (tự động bỏ qua hoa/thường)
#   từ khóa => bí danh    Đặt bí danh hiển thị cho từ khóa
#   [tên nhóm]        Dòng đầu tiên của nhóm, đặt tên hiển thị cho cả nhóm
#   +từ khóa          Từ bắt buộc, tất cả từ bắt buộc phải khớp mới tính
#   !từ khóa          Từ loại trừ, nếu khớp thì loại bỏ tin đó (chỉ áp dụng cho nhóm hiện tại)
#   @số               Giới hạn số tin tối đa hiển thị cho nhóm này
#
# Ưu tiên tên hiển thị:
#   1. Có tên nhóm → hiển thị tên nhóm
#   2. Không có tên nhóm → ghép các bí danh (nối bằng " / ")
#   3. Không có bí danh → hiển thị từ khóa gốc
#
#
# ┌─────────────────────────────────────────────────────────────┐
# │                  Cách Dùng Cơ Bản (Dành Cho Người Mới)      │
# └─────────────────────────────────────────────────────────────┘
#
# 1. Đơn giản nhất: viết thẳng từ khóa
#    ────────────────────
#    Vingroup
#
#    Kết quả: khớp tất cả tin có chứa "Vingroup"
#
#
# 2. Nhiều từ khóa trong một nhóm
#    ────────────────────
#    Vingroup
#    Vinhomes
#    Phạm Nhật Vượng
#
#    Kết quả: khớp tin có "Vingroup" HOẶC "Vinhomes" HOẶC "Phạm Nhật Vượng",
#             hiển thị tên "Vingroup / Vinhomes / Phạm Nhật Vượng"
#
#
# 3. Đặt tên cho nhóm (khuyến nghị)
#    ────────────────────
#    [Vingroup]
#    Vingroup
#    Vinhomes
#    Phạm Nhật Vượng
#
#    Kết quả: như trên, nhưng tên hiển thị là "Vingroup" (gọn hơn)
#
#
# ┌─────────────────────────────────────────────────────────────┐
# │                  Cách Dùng Nâng Cao (Tùy Chọn)              │
# └─────────────────────────────────────────────────────────────┘
#
# 4. Dùng biểu thức chính quy để khớp nhiều từ (một dòng gọn)
#    ────────────────────
#    /Vingroup|Vinhomes|Phạm Nhật Vượng/ => Vingroup
#
#    Kết quả: khớp tin có một trong ba từ trên, hiển thị là "Vingroup"
#    Giải thích: dùng | để phân cách nhiều từ, => để đặt tên hiển thị
#
#    💡 Không rành regex? Hỏi AI:
#       "Viết cho tôi một biểu thức chính quy khớp văn bản chứa 'Vingroup'
#        hoặc 'Vinhomes' hoặc 'Phạm Nhật Vượng',
#        định dạng yêu cầu: /regex/ => tên hiển thị"
#
#
# 5. Khớp chính xác từ tiếng Anh (tránh khớp nhầm)
#    ────────────────────
#    /\bAI\b/i => AI
#
#    Giải thích: \b là ranh giới từ, tránh khớp "AI" trong "MAIL"
#                /i là bỏ qua hoa/thường, khớp cả "ai", "AI", "Ai"
#
#
# 6. Loại trừ nội dung cụ thể
#    ────────────────────
#    [Apple]
#    Apple
#    !táo
#    !vườn
#
#    Kết quả: khớp "Apple" nhưng loại tin có chứa "táo" hoặc "vườn"
#
#
# 7. Giới hạn số tin hiển thị
#    ────────────────────
#    [Tin công nghệ]
#    công nghệ
#    @5
#
#    Kết quả: hiển thị tối đa 5 tin khớp
#
#
# 8. Bắt buộc chứa nhiều từ cùng lúc
#    ────────────────────
#    +Apple
#    +ra mắt
#
#    Kết quả: phải đồng thời chứa "Apple" VÀ "ra mắt" mới khớp
#
# ───────────────────────────────────────────────────────────────

[WORD_GROUPS]
